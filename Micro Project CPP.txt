        	}
			
	        if(s=="digital technique" || s=="dt")	
			    cout<<"				Ulhas Deshmukh Sir";
			    
			else if(s=="database management system" || s=="dbms")
			    cout<<"				Akshay Pakhare Sir";
			
			else if(s=="computer graphics" || s=="cgr")

			}
    }
